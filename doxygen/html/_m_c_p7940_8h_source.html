233 template <typename T>
234 uint8_t I2C_read(const uint8_t address, T& value) const {
245 uint8_t i{0}; // return number of bytes read
246 Wire.beginTransmission(MCP7940_ADDRESS); // Address the I2C device
247 Wire.write(address); // Send register address to read from
248 if (Wire.endTransmission() == 0) { // Close transmission and check error code
249 Wire.requestFrom(MCP7940_ADDRESS, sizeof(T)); // Request a block of data
250 uint8_t* bytePtr = (uint8_t*)&value; // Declare pointer to start of structure
251 for (i = 0; i < sizeof(T); i++) { // Loop for each byte to be read
252 *bytePtr++ = Wire.read(); // Read a byte
253 } // of for-next each byte
254 } // if-then success
255 return i; // return number of bytes read
256 } // end of template method "I2C_read"
257 template <typename T>
258 uint8_t I2C_write(const uint8_t address, const T& value) const {
268 Wire.beginTransmission(MCP7940_ADDRESS); // Address the I2C device
269 Wire.write(address); // Send register address to read from
270 Wire.write((uint8_t*)&value, sizeof(T)); // write the data
271 uint8_t i = Wire.endTransmission(); // close transmission and save status
272 if (i == 0) i = sizeof(T); // return number of bytes on success
273 return i; // return the number of bytes written
274 } // end of template method "I2C_write()"
